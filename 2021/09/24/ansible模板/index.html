ansible模板

Created2021-09-24|Updated2021-09-24
|Post View:

group_vars/all.yml

ansible_ssh_user: ‘root’
ansible_ssh_pass: ‘Laishui@YT’
ansible_ssh_port: ‘22’
tomcat_home: /opt/tomcat-gateway-uat
tomcat_home2: /opt/tomcat-queue

ansible.cfg
[defaults]
inventory = ./hosts
deprecation_warnings=False
host_key_checking = false

tomcat_deploy.yml

  • name: stop tomcat
    action: shell /bin/shutdown.sh

    忽略原本并没有 tomcat 运行的错误提示

    ignore_errors: yes
  • name: del ROOT dir
    shell: rm -rf /webapps/frontgateway
    ignore_errors: yes
  • name: del old war
    shell: rm -f /webapps/frontgateway.war
    ignore_errors: yes
  • name: copy war
    shell: tar -zxvf /opt/packages/frontgateway.tgz -C /webapps/
  • name: start tomcat
    shell: chdir= /bin nohup ./startup.sh &

gateway.yaml

  • name: deploy tomcat-gateway
    hosts: tomcat-gateway
    gather_facts: true
    remote_user: root
    tasks:
    • name: “backup war”
      shell: cp /webapps/frontgateway.war /opt/backup/gateway_$(date +”%Y%m%d%H%M%S”).war
      ignore_errors: yes
    • name: del old war
      shell: rm -f /webapps/frontgateway.war
      ignore_errors: yes
    • include: tomcat_deploy.yml
    • name: sleep 40s
      shell: sleep 40s
  • name: deploy tomcat-gateway
    hosts: tomcat-gateway
    tasks:
    • name: tomcat status
      shell: curl http://192.168.20.6:8082/frontgateway/ &>/dev/null && echo YES || echo NO
      ignore_errors: True
      register: is_tomcat_success
    • include: tomcat_deploy.yml
      when: is_tomcat_success.stdout == “YES”
