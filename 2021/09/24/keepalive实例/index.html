! Configuration File for keepalived

global_defs { # 接收到较低优先级广播后第二组免费 ARP 的延迟
vrrp_garp_lower_prio_delay 10 # 收到的低优先级广播后，一次发送的免费 ARP 数量
vrrp_garp_lower_prio_repeat 3
}

vrrp_instance VI_1 { # 使用单播通信,默认是组播通信
unicast_src_ip 172.16.20.29
unicast_peer {
172.16.20.28
172.16.20.27
} # 初始化状态 设为 BACKUP 恢复后不抢占
state BACKUP
interface ens224 # 虚拟路由标识，此 ID 要与 Backup 配置一致
virtual_router_id 51 # 默认启动优先级，要比 Backup 大点
priority 90
advert_int 5
nopreempt # 非抢占
#preempt_delay 600 # 抢占延迟 10min
garp_master_delay 30 #升级成 master 后延时 30s 再发送一次 arp 刷新请求
garp_master_refresh 30 # 刷新免费 ARP 的最小时间间隔，默认不刷新
authentication {
auth_type PASS
auth_pass abcd1234
}
virtual_ipaddress { # 虚拟 ip 地址
112.17.39.87/28 dev ens192
183.134.218.132/28 dev ens192
}
virtual_routes { # 虚拟路由，当 IP 漂过来之后需要添加的路由信息
default via 183.134.218.129 dev ens192
} # 状态切换到 master 执行脚本
#notify_master “/etc/keepalived/change_master_1.sh” # 状态切换到 backup 执行脚本
#notify_backup “/etc/keepalived/change_backup_1.sh”
}

lb_kind DR # 同一 IP 的连接 60 秒内被分配到同一台真实服务器
persistence_timeout 50
protocol TCP
