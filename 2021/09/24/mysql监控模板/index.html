mysql监控模板

Created2021-09-24|Updated2021-09-24
|Post View:

mysq 连接数： max(max_over_time(mysql_global_status_threads_connected{instance=“172._”}[1m]) or mysql_global_status_threads_connected{instance=“172.“} )
mysql_global_status_max_used_connections{instance=~”172.“}
mysql_global_variables_max_connections{instance=~”172._”}
mysql 正常运行时间：mysql_global_status_uptime{instance=”$host”}
current QPS : rate(mysql_global_status_queries{instance=”$host”}[$interval]) or irate(mysql_global_status_queries{instance=”$host”}[5m])
innoDB Buffer Pool Size: mysql_global_variables_innodb_buffer_pool_size{instance=”$host”}

mysql client thread activity 客户端线程活动： max_over_time(mysql_global_status_threads_connected{instance=”$host”}[$interval]) or
max_over_time(mysql_global_status_threads_connected{instance=”$host”}[5m])
max_over_time(mysql_global_status_threads_running{instance=”$host”}[$interval]) or
max_over_time(mysql_global_status_threads_running{instance=”$host”}[5m])
avg_over_time(mysql_global_status_threads_running{instance=”$host”}[$interval]) or
avg_over_time(mysql_global_status_threads_running{instance=”$host”}[5m])

mysql questions: rate(mysql_global_status_questions{instance=”$host”}[$interval]) or irate(mysql_global_status_questions{instance=”$host”}[5m])
mysql thread cache: mysql_global_variables_thread_cache_size{instance=”$host”}
mysql_global_status_threads_cached{instance=”$host”}
rate(mysql_global_status_threads_created{instance=”$host”}[$interval]) or irate(mysql_global_status_threads_created{instance=”$host”}[5m])

mysql temporary objects: rate(mysql_global_status_created_tmp_tables{instance=”$host”}[$interval]) or irate(mysql_global_status_created_tmp_tables{instance=”$host”}[5m])
rate(mysql_global_status_created_tmp_disk_tables{instance=”$host”}[$interval]) or irate(mysql_global_status_created_tmp_disk_tables{instance=”$host”}[5m])
rate(mysql_global_status_created_tmp_files{instance=”$host”}[$interval]) or irate(mysql_global_status_created_tmp_files{instance=”$host”}[5m])

mysql select types: rate(mysql_global_status_select_full_join{instance=”$host”}[$interval]) or irate(mysql_global_status_select_full_join{instance=”$host”}[5m])

mysql sorts 排序：rate(mysql_global_status_sort_rows{instance=”$host”}[$interval]) or irate(mysql_global_status_sort_rows{instance=”$host”}[5m])
mysql slow queries慢查询： rate(mysql_global_status_slow_queries{instance=”$host”}[$interval]) or irate(mysql_global_status_slow_queries{instance=”$host”}[5m])

mysql aborted connections 中止连接： rate(mysql_global_status_aborted_connects{instance=”$host”}[$interval]) or irate(mysql_global_status_aborted_connects{instance=”$host”}[5m])
rate(mysql_global_status_aborted_clients{instance=”$host”}[$interval]) or irate(mysql_global_status_aborted_clients{instance=”$host”}[5m])

mysql table locks 锁表： rate(mysql_global_status_table_locks_immediate{instance=”$host”}[$interval]) or irate(mysql_global_status_table_locks_immediate{instance=”$host”}[5m])
rate(mysql_global_status_table_locks_waited{instance=”$host”}[$interval]) or irate(mysql_global_status_table_locks_waited{instance=”$host”}[5m])

mysql network traffic 网络流量： rate(mysql_global_status_bytes_received{instance=”$host”}[$interval]) or irate(mysql_global_status_bytes_received{instance=”$host”}[5m])
rate(mysql_global_status_bytes_sent{instance=”$host”}[$interval]) or irate(mysql_global_status_bytes_sent{instance=”$host”}[5m])

mysql network usage hourly 每小时网络使用情况： increase(mysql_global_status_bytes_received{instance=”$host”}[1h])
increase(mysql_global_status_bytes_sent{instance=”$host”}[1h])

mysql 内存概述: node_memory_MemTotal_bytes{instance=”$host”}
mysql_global_status_innodb_page_size{instance=”$host”} * on (instance) mysql_global_status_buffer_pool_pages{instance=”$host”,state=”data”}
mysql_global_variables_innodb_log_buffer_size{instance=”$host”}
mysql_global_variables_innodb_additional_mem_pool_size{instance=”$host”}
mysql_global_status_innodb_mem_dictionary{instance=”$host”}
mysql_global_variables_key_buffer_size{instance=”$host”}
mysql_global_variables_query_cache_size{instance=”$host”}
mysql_global_status_innodb_mem_adaptive_hash{instance=”$host”}
mysql_global_variables_tokudb_cache_size{instance=”$host”}

top command counters: topk(5, rate(mysql_global_status_commands_total{instance=”$host”}[$interval])>0) or topk(5, irate(mysql_global_status_commands_total{instance=”$host”}[5m])>0)
top command counters hourly: topk(5, increase(mysql_global_status_commands_total{instance=”$host”}[1h])>0)

mysql handlers: rate(mysql_global_status_handlers_total{instance=”$host”, handler!“commit|rollback|savepoint.*|prepare”}[$interval]) or irate(mysql_global_status_handlers_total{instance=”$host”, handler!“commit|rollback|savepoint.*|prepare”}[5m])
mysql transaction handlers: rate(mysql_global_status_handlers_total{instance=”$host”, handler=“commit|rollback|savepoint._|prepare”}[$interval]) or irate(mysql_global_status_handlers_total{instance=”$host”, handler=“commit|rollback|savepoint._|prepare”}[5m])

mysql file openings: rate(mysql_global_status_opened_files{instance=”$host”}[$interval]) or irate(mysql_global_status_opened_files{instance=”$host”}[5m])
mysql open files: mysql_global_status_open_files{instance=”$host”}
mysql_global_variables_open_files_limit{instance=”$host”}
mysql_global_status_innodb_num_open_files{instance=”$host”}

mysql table open cache status: rate(mysql_global_status_opened_tables{instance=”$host”}[$interval]) or irate(mysql_global_status_opened_tables{instance=”$host”}[5m])
rate(mysql_global_status_table_open_cache_hits{instance=”$host”}[$interval]) or irate(mysql_global_status_table_open_cache_hits{instance=”$host”}[5m])
rate(mysql_global_status_table_open_cache_misses{instance=”$host”}[$interval]) or irate(mysql_global_status_table_open_cache_misses{instance=”$host”}[5m])
rate(mysql_global_status_table_open_cache_overflows{instance=”$host”}[$interval]) or irate(mysql_global_status_table_open_cache_overflows{instance=”$host”}[5m])
(rate(mysql_global_status_table_open_cache_hits{instance=”$host”}[$interval]) or irate(mysql_global_status_table_open_cache_hits{instance=”$host”}[5m]))/((rate(mysql_global_status_table_open_cache_hits{instance=”$host”}[$interval]) or irate(mysql_global_status_table_open_cache_hits{instance=”$host”}[5m]))+(rate(mysql_global_status_table_open_cache_misses{instance=”$host”}[$interval]) or irate(mysql_global_status_table_open_cache_misses{instance=”$host”}[5m])))

mysql open tables: mysql_global_status_open_tables{instance=”$host”}
mysql_global_variables_table_open_cache{instance=”$host”}

mysql table definetion cache: mysql_global_status_open_table_definitions{instance=”$host”}
mysql_global_variables_table_definition_cache{instance=”$host”}
rate(mysql_global_status_opened_table_definitions{instance=”$host”}[$interval]) or irate(mysql_global_status_opened_table_definitions{instance=”$host”}[5m])
