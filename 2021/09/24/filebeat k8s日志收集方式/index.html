—- filbeat.yaml 模板—————
filebeat.inputs:

  • type: log
    paths:
    • path
      tags: [“xxpro”,”plancenterpre”]

name: 192.168.1.237
processors:

  • drop_fields:
    fields: [“agent.ephemeral_id”, “agent.version”, “ecs.version”, “log.offset”, “agent.id”]

output.logstash:
hosts: [“xxxxx:5044”]

–指定 tag,正则切割以及索引

  • type: log
    paths: - /opt/logs/uat/cactus/*.log
    tags: [“aliuat”,”doct”]
    multiline:
    pattern: ‘^\d{4}-\d{2}-\d{2}’
    negate: true
    match: after

    filetype: other

​

— docker-entrypoint.sh –容器初始化脚本
​

#!/bin/bash
ENV=${ENV:-“plan”}
FILETYPE=${FILETYPE:-“plan”}
#PROJECT=${project:-“xxpre”}
PROJECT=${PROJECT:-“xxpre”}
PROJ_NAME=${PROJ_NAME:-“no-define”}
MULTILINE=${MULTILINE:-“^\d{4}-\d{2}-\d{2}”}
cat > /etc/filebeat.yaml << EOF
filebeat.inputs:

  • type: log
    fields_under_root: true
    fields:
    topic: logm-${PROJ_NAME}
    filetype: $FILETYPE
    paths:
    • /logm/*.log
    • /logm//.log
    • /logm///*.log
    • /logm////.log
    • /logm/////*.log
      scan_frequency: 120s
      tags: [“$PROJECT”,”$ENV”]
      exclude_files: [‘perf.log’,’20’]
      max_bytes: 10485760
      multiline.pattern: ‘$MULTILINE’
      multiline.negate: true
      multiline.match: after
      multiline.max_lines: 3000
  • type: log
    fields_under_root: true
    fields:
    topic: logu-${PROJ_NAME}
    paths: - /logu/.log - /logu//.log - /logu///.log - /logu////.log - /logu/////.log - /logu//////*.log
    #output.kafka:

hosts: [“10.100.243.217:9092”,”10.108.213.213:9092”,”10.101.165.156:9092”]

topic: topic-$ENV

version: 2.0.0

required_acks: 0

max_message_bytes: 10485760

output.logstash:
hosts: [“xxxxxx:5044”]

EOF
set -xe
if [[“$1” == “”]]; then
exec /usr/local/filebeat/filebeat -c /etc/filebeat.yaml
else
exec “$@”
fi
